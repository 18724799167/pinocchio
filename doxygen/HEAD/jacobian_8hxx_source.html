71  /* Return the jacobian of the output frame attached to joint <jointId> in the
72  world frame or in the local frame depending on the template argument. The
73  function computeJacobians should have been called first. */
74  template<bool localFrame>
75  void getJacobian(const Model & model,
76  const Data & data,
77  const Model::JointIndex jointId,
78  Data::Matrix6x & J)
79  {
80  assert( J.rows() == data.J.rows() );
81  assert( J.cols() == data.J.cols() );
82 
83  const SE3 & oMjoint = data.oMi[jointId];
84  int colRef = nv(model.joints[jointId])+idx_v(model.joints[jointId])-1;
85  for(int j=colRef;j>=0;j=data.parents_fromRow[(Model::Index)j])
86  {
87  if(! localFrame ) J.col(j) = data.J.col(j);
88  else J.col(j) = oMjoint.actInv(Motion(data.J.col(j))).toVector();
89  }
90  }

122  inline const Data::Matrix6x &
123  jacobian(const Model & model, Data & data,
124  const Eigen::VectorXd & q,
125  const Model::JointIndex jointId)
126  {
127  data.iMf[jointId].setIdentity();
128  for( Model::JointIndex i=jointId;i>0;i=model.parents[i] )
129  {
130  JacobianForwardStep::run(model.joints[i],data.joints[i],
131  JacobianForwardStep::ArgsType(model,data,q));
132  }
133 
134  return data.J;
